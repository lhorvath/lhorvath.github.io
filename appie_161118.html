def save_obj(obj, name ):
    with open('obj/'+ name + '.pkl', 'wb') as f:
        pickle.dump(obj, f, pickle.HIGHEST_PROTOCOL)

def load_obj(name ):
    with open('obj/' + name + '.pkl', 'rb') as f:
        return pickle.load(f)

# 11 document-feature matrixes directly from mediacloud api
# with 11 respective word lists as dictionaries
m = load_obj('m')

# urls
u = load_obj('u')

# date stamps
dates = load_obj('dates')

# 'reconstruct' texts from document-feature matrixes
# the api call returned DFMs and lists of word labels, however, gensim has no way of using--
# --externally provided labels lists (at least I don't know how)
# thus the idea is to map word lsits to DTMs and reconstruct 'pseudo-texts' so that
# gensim can do its own vocab building in the background.

# document ids: varying numbers of articles nested within 11 groups (these are the 11 api calls)
# when articles were collected through the api
article_ids = [list(m[i]['word_matrix'].keys()) for i in range(len(m))]

# first create, for each article j, a list of words (their numerical codes)
# within each group, i
# these words are stored as dictionary keys
articles_coded = copy.deepcopy(article_ids) #  copying for structure 

for i in range(len(article_ids)):
    for j in range(len(article_ids[i])):
        articles_coded[i][j] = list(m[i]['word_matrix'][article_ids[i][j]].keys())

# repeat, in each article j, the word (its numerical code) as many times as indicated in the DFM
# across all groups, i
# the frequency (repetition) of each word is indicated as dictionary values
# first get a list of these values
articles_rep = copy.deepcopy(article_ids) #  copying for structure 

for i in range(len(article_ids)):
    for j in range(len(article_ids[i])):
        articles_rep[i][j] = list(m[i]['word_matrix'][article_ids[i][j]].values())

# then map word to rep value
for i in range(len(article_ids)):
    for j in range(len(article_ids[i])):
        articles_coded[i][j] = np.repeat(articles_coded[i][j], articles_rep[i][j])

# np array to list for further use
articles_decoded = copy.deepcopy(articles_coded) #  copying for structure 

for i in range(len(article_ids)):
    for j in range(len(article_ids[i])):
        articles_decoded[i][j] = np.ndarray.tolist(articles_decoded[i][j])

# finally map words' numerical code to word label
for group in range(len(article_ids)):
    for article in range(len(article_ids[group])):
        for word in range(len(articles_decoded[group][article])): 
            articles_decoded[group][article][word] = m[group]['word_list'][int(articles_decoded[group][article][word])][0]
 
# now let gensim create its dictionary
dictionary = [corpora.Dictionary(articles_decoded[group]) for group in range(len(articles_decoded))]

corpus = copy.deepcopy(article_ids)

for group in range(len(corpus)):
    for article in range(len(corpus[group])):
        corpus[group][article] = dictionary[group].doc2bow(articles_decoded[group][article])

########## lda models. 
# test multiple values of k against semantic coherence

start_k = 2
end_k = 6
step_k = 1

# append into nested list (k runs from 2 to 5 in each group)
results_k = []

for group in range(len(article_ids)):
    results_sub = []
    for k in range(start_k, end_k, step_k):
        #print('Fit k = %s' % k)
        lda = LdaModel(corpus[group], num_topics=k, iterations=100,
                   alpha = 'auto', id2word=dictionary[group], 
                   random_state=12345)
        
        cm = CoherenceModel(model=lda, 
                        texts=articles_decoded[group], 
                        dictionary=dictionary[group], 
                        coherence='c_v',
                        topn=10)
        
        coherence = cm.get_coherence()
        #print('Coherence = %s' % coherence)

        results_sub.append([k, coherence])
    
    results_k.append([results_sub])

# max_c stores the recommended k value based on semantic coherence
max_c = []
klist = [2,3,4,5]

for group in range(len(article_ids)):
    c = []
    for row in range(len(klist)):
        c.append([results_k[group][0][row][1]])
    
    max_c.append(klist[np.argmax(c)])

# fit lda models using the best k
lda = []

for group in range(len(article_ids)):
    mod = LdaModel(corpus[group], num_topics = max_c[group], id2word = dictionary[group], random_state = 12345)
    lda.append([mod])

def top_documents(content, topic_id, theta, n = 10):
    ''' Takes a topic id and the full document-topic 
        matrix (theta) and returns the indices for the 
        top n documents. '''
    
    x = theta[topic_id,:].todense()
    idx = np.argpartition(x[0,:], -n)[0,-n:]
    
    # Get sorted IDs
    idx_sorted = idx[0, np.argsort(-x[0,idx])].tolist()[0]
    
    # Find and return the top documents
    return [row for i, row in enumerate(content) if i in set(idx_sorted)]
