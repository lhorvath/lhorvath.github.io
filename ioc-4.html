# bivariate plots /1: x & y continuous variables
# plot() function opens dynamic environment -> you can add layers to it
plot(hsb2$math, hsb2$science, 
     col = "magenta", pch = 22, 
     main = "Bivariate relationship",
     ylab = "Science",
     xlab = "Maths")
points(hsb2$math, hsb2$socst, 
       col = "blue", pch = 22, 
       main = "Bivariate relationship",
       ylab = "Science",
       xlab = "Maths")

ggplot demonstration

# install.packages(ggplot2)
library(ggplot2)

ggplot(hsb2, aes(x=math, y=science)) + 
  geom_point(aes(color=as.factor(female))) + # try without as.factor()
  theme_bw()

hsb2$gender[hsb2$female==1] <- "Women"
hsb2$gender[hsb2$female==0] <- "Men"

ggplot(hsb2, aes(x=as.factor(gender), y=science)) + 
  geom_boxplot(col = "blue", fill = "white") + 
  coord_flip() +
  facet_grid(ses ~ .)
